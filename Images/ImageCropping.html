working on a single plane how to brighten

brighten <- function(image, howmuch = 0.75) {
    return((1 - howmuch) * image + howmuch)
}

framer <- function(image, width) {
    size <- dim(image)
    image <- cbind(brighten(image[, width:1]), image, brighten(image[, size[2]:(size[2] - 
        width)]))
    image <- rbind(brighten(image[width:1, ]), image, brighten(image[size[1]:(size[1] - 
        width), ]))
    return(image)
}

show <- function(image) {
    size <- dim(image)
    canvas(x = c(1, size[2]), y = c(1, size[1]), asp = 1)
    rasterImage(image, 1, 1, size[2], size[1])
}

red = framer(puppy[, , 1], 20)
green = framer(puppy[, , 2], 20)
blue = framer(puppy[, , 3], 20)
show(array(c(red, green, blue), dim = c(dim(red), 3)))
