outlier <- function(x) {
    scope1 <- quantile(x, c(0.25))
    scope2 <- quantile(x, c(0.75))
    bounds <- 1.5 * (scope2 - scope1)
    upper <- bounds + scope2  # 10
    lower <- scope1 - bounds  # 2
    # Another way
    res <- x < lower | x > upper

    # One way
    res <- rep(NA, length(x))
    for (k in 1:length(x)) {
        res[k] <- x[k] > upper | x[k] < lower
    }
    return(res)
}

digitToWord <- function(digits, words) {
    res <- character(length(digits))
    for (k in 1:length(digits)) {
        res[k] = words[digits[k]]
    }
    return(res)
}

digitToWord(c(1, 2, 3, 4), c("un", "deux", "trois", "quatre"))

## [1] "un"     "deux"   "trois"  "quatre"

lettersMatch <- function(pattern, words) {
    grepPattern <- paste("^", pattern, "$", sep = "")
    Match <- grepl(grepPattern, words)
    return(Match)
}

lettersMatch("mere", "grandmere")

## [1] FALSE

lettersMatch("ing", "Programming")

## [1] FALSE

lettersMatch("excel", "excellence")

piSeries <- function(n) {
    num <- -((-1)^(1:n))
    denom <- seq(1, 2 * n, by = 2)
    return(4 * sum((num/denom)))
}

## [1] 3.132

piSeries(1000)

piTerms <- function(n) {
    num <- -((-1)^(1:n))
    denom <- seq(1, 2 * n, by = 2)
    return(cumsum(4 * (num/denom)) - pi)
}

howCloseToPi2 <- function(n) {
    num <- 1:n
    # apply the piSeries function to each 'num'
    plot(sapply(num, piSeries, simplify = TRUE))
}

howCloseToPi2(100)


howCloseToPi2(1000)

# Hence, really you need 601 terms to approximate pi.

# I would say aproximately 1000 billion. The sum converges to pi as n goes
# to infinity. So you would need a large number of n.

randomApproxToPi <- function(n) {
    x <- runif(n)
    y <- runif(n)
    distance <- x^2 + y^2
    return(4 * sum(distance <= 1)/n)
}

randomApproxToPi(150)

## [1] 3.307

randomApproxToPi(1000)

## [1] 3.204
