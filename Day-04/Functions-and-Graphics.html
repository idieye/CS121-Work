# How many odds
oddcount <- function(x) {
    sum((x%%2 == 0) == FALSE)
}

# oddcount Test

## [1] 4

## [1] 6

LawOfcosine <- function(a, b, theta) {
    c = sqrt(a^2 + b^2 - 2 * a * b * cos(theta))
    return(c)
}

# Find angle
angleFinder <- function(a, b, c) {
    theta = acos((c^2 - a^2 - b^2)/(-2 * a * b))
    return(theta)
}

angleFinder(3, 4, 5)

## [1] 1.571

angleFinder(13, 84, 85)

## [1] 1.571

angleFinder(3, 2, 2.125)

# Circles
Drawcircle <- function(x, y, r, color) {
    angle <- seq(0, 2 * pi, length = 100)
    xpoints <- x + r * cos(angle)
    ypoints <- y + r * sin(angle)
    polygon(xpoints, ypoints, col = color, border = NULL)
}

# Overlapping circles
