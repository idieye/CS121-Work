# crossword helper

crosswordhelper <- function(words, pattern) {

    matched <- words[grep(pattern, words)]
    return(matched)
}

crosswordPattern <- function(length, ...) {
    stuff <- list(...)
    blank <- c("^", rep(".", length), "$")
    for (k in 1:length(stuff)) {
        blank[stuff[[k]] + 1] <- names(stuff[k])
    }
    pattern <- paste(blank, collapse = "")
    return(crosswordhelper(words, pattern))
}
